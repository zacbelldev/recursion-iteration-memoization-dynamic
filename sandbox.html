function fibonacci(num) {
    // iteration method
    var a = 0, b = 1, f = 1;
    var s = "1, ";
    for (var i = 2; i <= num; i++) {
        f = a + b;
        a = b;
        b = f;
        s += f + ", ";
    }
    if (num < 0)
        return ;
    else if (num == 0)
        return 0;
    else
        return (document.getElementById("iterative-output")
        .innerHTML = s.replace(/,\s*$/, ""));
}
