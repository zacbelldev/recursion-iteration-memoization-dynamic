function recursion_fibonacci(num) {
    if (num < 0)
        return (document.getElementById("output").innerHTML = 
        "Error");
    else if (num < 2)
        return (document.getElementById("output").innerHTML = 
        num);
    else
        return (document.getElementById("output").innerHTML = 
        recursion_fibonacci(num - 1) + recursion_fibonacci(num - 2));
}
